518 log_appender_t
519 log_add_appender (log_appender_fn appender_fp, log_level_t level, void* udata)
520 {
521  // Initialize logger if neccesary
522  log_try_init();
523 
524  // Check if there is space for a new appender.
525  LOG_ASSERT(log_appender_count < LOG_MAX_APPENDERS);
526 
527  // Ensure level is valid
528  LOG_ASSERT(level >= 0 && level < LOG_LEVEL_COUNT);
529 
530  // Iterate through appender array and find an empty slot.
531  for (int i = 0; i < LOG_MAX_APPENDERS; i++)
532  {
533  if (NULL == log_appenders[i].appender_fp)
534  {
535  log_appender_data_t* appender = &log_appenders[i];
536 
537  // Store and enable appender
538  appender->appender_fp = appender_fp;
539  appender->log_level = level;
540  appender->udata = udata;
541  appender->level = LOG_LEVEL_INFO;
542  appender->enabled = true;
543  appender->colors = false;
544  appender->level = true;
545  appender->timestamp = false;
546  appender->file = false;
547  appender->func = false;
548  appender->lock_fp = NULL;
549  appender->lock_udata = NULL;
550 
551  strncpy(appender->time_fmt, LOG_TIME_FMT, LOG_TIME_FMT_LEN);
552 
553  log_appender_count++;
554 
555  return (log_appender_t)i;
556  }
557  }
558 
559  // This should never happen
560  LOG_ASSERT(false);
561  return 0;
562 }

806 void
807 log_write (log_level_t level, const char* file, unsigned line,
808  const char* func, const char* fmt, ...)
809 {
810  // Ensure logger is initialized
811  log_try_init();
812 
813  // Only write entry if there are registered appenders and the logger is
814  // enabled
815  if (0 == log_appender_count || !log_enabled)
816  {
817  return;
818  }
819 
820  // Ensure valid log level
821  LOG_ASSERT(level < LOG_LEVEL_COUNT);
822 
823  for (log_appender_t i = 0; i < LOG_MAX_APPENDERS; i++)
824  {
825  log_appender_data_t* appender = &log_appenders[i];
826 
827  if (!log_appender_enabled(i))
828  continue;
829 
830  if (log_appenders[i].log_level <= level)
831  {
832  char entry_str[LOG_ENTRY_LEN + 1]; // Ensure there is space for
833  // null char
834 
835  entry_str[0] = '\0'; // Ensure the entry is null terminated
836 
837  // Append a timestamp
838  if (appender->timestamp)
839  {
840  log_append_timestamp(entry_str, appender->time_fmt);
841  }
842 
843  // Append the logger level
844  if (appender->level)
845  {
846  log_append_level(entry_str, level, appender->colors);
847  }
848 
849  // Append the filename/line number
850  if (appender->file)
851  {
852  log_append_file(entry_str, file, line);
853  }
854 
855  // Append the function name
856  if (appender->func)
857  {
858  log_append_func(entry_str, func, appender->colors);
859  }
860 
861  // Append the log message
862  char msg_str[LOG_MSG_LEN];
863 
864  va_list args;
865  va_start(args, fmt);
866  vsnprintf(msg_str, sizeof(msg_str), fmt, args);
867  va_end(args);
868 
869  strncat(entry_str, msg_str, LOG_MSG_LEN);
870  strcat(entry_str, "\n");
871 
872  // Locks the appender
873  if (NULL != appender->lock_fp)
874  {
875  appender->lock_fp(true, appender->lock_udata);
876  }
877 
878  appender->appender_fp(entry_str, appender->udata);
879 
880  // Unlocks the appender
881  if (NULL != appender->lock_fp)
882  {
883  appender->lock_fp(false, appender->lock_udata);
884  }
885  }
886  }
887 }
