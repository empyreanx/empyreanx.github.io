433{

486pl_id_t
487pl_add_appender (pl_appender_fn appender_fp, pl_level_t level, void* udata)
488{
489 // Initialize logger if neccesary
490 pl_try_init();
491
492 // Check if there is space for a new appender.
493 PL_ASSERT(pl_appender_count < PL_MAX_APPENDERS);
494
495 // Ensure level is valid
496 PL_ASSERT(level >= 0 && level < PL_LEVEL_COUNT);
497
498 // Iterate through appender array and find an empty slot.
499 for (int i = 0; i < PL_MAX_APPENDERS; i++)
500 {
501 if (NULL == pl_appenders[i].appender_fp)
502 {
503 pl_appender_t* appender = &pl_appenders[i];
504
505 // Store and enable appender
506 appender->appender_fp = appender_fp;
507 appender->log_level = level;
508 appender->udata = udata;
509 appender->level = PL_LEVEL_INFO;
510 appender->enabled = true;
511 appender->colors = false;
512 appender->level = true;
513 appender->timestamp = false;
514 appender->file = false;
515 appender->func = false;
516 appender->lock_fp = NULL;
517 appender->lock_udata = NULL;
518
519 strncpy(appender->time_fmt, PL_TIME_FMT, PL_TIME_FMT_LEN);
520
521 pl_appender_count++;
522
523 return (pl_id_t)i;
524 }
525 }
526
527 // This should never happen
528 PL_ASSERT(false);
529 return 0;
530}

786void
787pl_write (pl_level_t level, const char* file, unsigned line,
788 const char* func, const char* fmt, ...)
789{
790 // Only write entry if there are registered appenders and the logger is
791 // enabled
792 if (0 == pl_appender_count || !pl_enabled)
793 {
794 return;
795 }
796
797 // Ensure valid log level
798 PL_ASSERT(level < PL_LEVEL_COUNT);
799
800 for (pl_id_t i = 0; i < PL_MAX_APPENDERS; i++)
801 {
802 pl_appender_t* appender = &pl_appenders[i];
803
804 if (!pl_appender_enabled(i))
805 continue;
806
807 if (pl_appenders[i].log_level <= level)
808 {
809 char entry_str[PL_ENTRY_LEN + 1]; // Ensure there is space for
810 // null char
811
812 entry_str[0] = '\0'; // Ensure the entry is null terminated
813
814 // Append a timestamp
815 if (appender->timestamp)
816 {
817 pl_append_timestamp(entry_str, appender->time_fmt);
818 }
819
820 // Append the logger level
821 if (appender->level)
822 {
823 pl_append_level(entry_str, level, appender->colors);
824 }
825
826 // Append the filename/line number
827 if (appender->file)
828 {
829 pl_append_file(entry_str, file, line, appender->colors);
830 }
831
832 // Append the function name
833 if (appender->func)
834 {
835 pl_append_func(entry_str, func, appender->colors);
836 }
837
838 // Append the log message
839 char msg_str[PL_MSG_LEN];
840
841 va_list args;
842 va_start(args, fmt);
843 vsnprintf(msg_str, sizeof(msg_str), fmt, args);
844 va_end(args);
845
846 strncat(entry_str, msg_str, PL_MSG_LEN);
847 strcat(entry_str, "\n");
848
849 // Locks the appender
850 if (NULL != appender->lock_fp)
851 {
852 appender->lock_fp(true, appender->lock_udata);
853 }
854
855 appender->appender_fp(entry_str, appender->udata);
856
857 // Unlocks the appender
858 if (NULL != appender->lock_fp)
859 {
860 appender->lock_fp(false, appender->lock_udata);
861 }
862 }
863 }
864}
