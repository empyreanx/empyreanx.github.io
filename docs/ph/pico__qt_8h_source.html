559 // Don't call this directly -- use `qt_array_fit` instead.
560 static void* qt_array_fit_impl(const void* array, int new_size, size_t element_size, void* mem_ctx)
561 {
562  (void)mem_ctx;
563 
564  QT_ARRAY_CANARY(array);
565 
566  // Double the old capacity, or pick at least 16 for the starting size.
567  // This helps unnecessarily numerous realloc's for low capacities when starting out.
568  int new_capacity = qt_max(2 * qt_array_capacity(array), qt_max(new_size, 16));
569 
570  QT_ASSERT(new_size <= new_capacity);
571 
572  // Total size of the header struct `qt_array_header_t` along with the size of all
573  // elements packed together in a single allocation.
574  size_t total_size = sizeof(qt_array_header_t) + new_capacity * element_size;
575 
576  qt_array_header_t* hdr;
577 
578  if (array)
579  {
580  // Realloc of the header isn't new.
581  // This expands the capacity.
582  hdr = (qt_array_header_t*)QT_REALLOC(qt_hdr(array), total_size, mem_ctx);
583  }
584 
585  else
586  {
587  // Create a new array if the pointer passed in was NULL, as NULL means an empty array.
588  hdr = (qt_array_header_t*)QT_MALLOC(total_size, mem_ctx);
589  hdr->size = 0;
590  }
591 
592  hdr->cookie = QT_ARRAY_COOKIE; // For sanity checks with `QT_ARRAY_CANARY`.
593  hdr->capacity = new_capacity;
594  hdr->data = (char*)(hdr + 1); // For debugging convenience.
595 
596  return (void*)(hdr + 1);
597 }

678 static void qt_node_insert(qt_t* qt, qt_node_t* node, const qt_rect_t* bounds, qt_value_t value)
679 {
680  QT_ASSERT(node);
681  QT_ASSERT(bounds);
682 
683  // The purpose of this function is to optimally fit the item into a subtree.
684  // This occurs when the item is no longer fully contained within a subtree,
685  // or the depth limit has been reached.
686 
687  // Checks to see if the depth limit has been reached. If it hasn't, try to
688  // fit the item into a subtree
689  if (node->depth + 1 < node->max_depth)
690  {
691  // Loop over child nodes
692  for (int i = 0; i < 4; i++)
693  {
694  // Check if subtree contains the bounds
695  if (qt_rect_contains(&node->bounds[i], bounds))
696  {
697  // If child node does not exist, then create it
698  if (!node->nodes[i])
699  {
700  node->nodes[i] = qt_node_create(qt,
701  node->bounds[i],
702  node->depth + 1,
703  node->max_depth);
704  }
705 
706  // Recursively try to insert the item into the subtree
707  qt_node_insert(qt, node->nodes[i], bounds, value);
708  return;
709  }
710  }
711  }
712 
713  // If none of the children fully contain the bounds, or the maximum depth
714  // has been reached, then the item belongs to this node
715  qt_array_push(qt->mem_ctx, node->items, (qt_item_t){ *bounds, value });
716 }

829  // If the area contains the the entire subtree, all items in the
830  // subtree match and are recursively added to the array
831  if (qt_rect_contains(area, &node->bounds[i]))
832  {
833  values = qt_node_all_values(qt, node->nodes[i], values);
834  }
835  else
836  {
837  // Otherwise, if the area intersects the bounds of the subtree,
838  // the subtree is recursively searched for items intersecting
839  // or contained within the area
840  if (qt_rect_overlaps(area, &node->bounds[i]))
841  {
842  values = qt_node_query(qt, node->nodes[i], area, values);
843  }
844  }
845  }
846  }
847 
848  return values;
849 }

863 {

865  unode->next = allocator->free_list;

871  qt_node_allocator_t* allocator = &qt->allocator;

877 static void qt_block_alloc(qt_t* qt)
878 {
879  // Allocate space for more nodes and add them to the free list.
880 
881  qt_node_allocator_t* allocator = &qt->allocator;
882 
883  const int block_count = PICO_QT_BLOCK_SIZE;
884 
885  qt_unode_t* nodes = (qt_unode_t*)QT_MALLOC(sizeof(qt_unode_t) * block_count, qt->mem_ctx);
886 
887  for (int i = 0; i < block_count; i++)
888  {
889  qt_freelist_push(qt, &nodes[i]);
890  }
891 
892  qt_array_push(qt->mem_ctx, allocator->blocks, nodes);
893 }

895 static qt_node_t* qt_node_alloc(qt_t* qt)
896 {
897  qt_node_allocator_t* allocator = &qt->allocator;
898 
899  if (!allocator->free_list)
900  {
901  qt_block_alloc(qt);
902  }
903 
904  // Pop a node off of the free list.
905  qt_unode_t* unode = qt_freelist_pop(qt);
906  QT_MEMSET(unode, 0, sizeof(qt_unode_t));
907 
908  qt_node_t* result = &unode->node;
909 
910  return result;
911 }

914 {
915  qt_freelist_push(qt, (qt_unode_t*)node);
