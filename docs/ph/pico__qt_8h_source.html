387 static void qt_array_resize(qt_array_t* array, int size)
388 {
389  QT_ASSERT(array);
390 
391  // No change if size is below the capacity
392  if (size < array->capacity)
393  {
394  array->size = size;
395  return;
396  }
397 
398  // Calculate new capacity
399  while (array->capacity <= size)
400  {
401  array->capacity += (array->capacity / 2) + 2;
402  }
403 
404  // Reallocate the array and set the new size
405  array->items = QT_REALLOC(array->items, array->capacity * sizeof(qt_item_t));
406  array->size = size;

424 static void qt_array_cat(qt_array_t* dst, const qt_array_t* src)
425 {
426  QT_ASSERT(dst);
427  QT_ASSERT(src);
428 
429  int total_capacity = dst->capacity + src->capacity;
430 
431  // Resize the array if dst capacity is less than the sum of the capacities
432  // This is the most likely case
433  if (dst->capacity < total_capacity)
434  {
435  dst->items = QT_REALLOC(dst->items, total_capacity * sizeof(qt_item_t));
436  dst->capacity = total_capacity;
437  }
438 
439  // Copy the contents of the src array onto the end of the dst array
440  memcpy(&dst->items[dst->size], src->items, src->size * sizeof(qt_item_t));
441 
442  // Increase the size to match
443  dst->size += src->size;
444 }

447 {
448  QT_ASSERT(array);
449  QT_ASSERT(index < array->size);
450 
451  int size = array->size;
452 
453  // Overwrites the item at the index with the item at the end of the array
454  // This is fast, but it changes the order of the array Fortunately, order
455  // doesn't matter in this case

491 {
492  qt_node_t* node = QT_MALLOC(sizeof(qt_node_t));
493 
494  memset(node, 0, sizeof(qt_node_t));
495 
496  node->depth = depth;
497 
498  // Calculate subdivided bounds
499  bounds.w /= 2.0f;
500  bounds.h /= 2.0f;
501 
502  // Calculates bounds of subtrees
503  node->bounds[0] = qt_make_rect(bounds.x,
504  bounds.y,
505  bounds.w,
506  bounds.h);
507 
508  node->bounds[1] = qt_make_rect(bounds.x + bounds.w,
509  bounds.y,
510  bounds.w,
511  bounds.h);
512 
513  node->bounds[2] = qt_make_rect(bounds.x,
514  bounds.y + bounds.h,
515  bounds.w,
516  bounds.h);
517 
518  node->bounds[3] = qt_make_rect(bounds.x + bounds.w,
519  bounds.y + bounds.h,
520  bounds.w,
521  bounds.h);

547 static void qt_node_insert(qt_node_t* node, const qt_rect_t* bounds, qt_value_t value)
548 {
549  QT_ASSERT(node);
550  QT_ASSERT(bounds);
551 
552  // Checks to see if item can possibly be inserted into subtree
553  if (node->depth + 1 < QT_MAX_DEPTH)
554  {
555  // Loop over child nodes
556  for (int i = 0; i < 4; i++)
557  {
558  // Check if child node contains the bounds
559  if (qt_rect_contains(&node->bounds[i], bounds))
560  {
561  // If max depth has not been reached and the bounds are fully
562  // contained within the child's bounds, then try to recursively
563  // fit the item into a node at a greater depth.
564 
565  // If child node does not exist, then create it
566  if (!node->nodes[i])
567  {
568  node->nodes[i] = qt_node_create(node->bounds[i], node->depth + 1);
569  }
570 
571  // The recursive call
572  qt_node_insert(node->nodes[i], bounds, value);
573  return;
574  }
575  }
576  }
577 
578  // If none of the children fully contain the bounds, or the maximum depth
579  // has been reached, then the item belongs to this node
580  qt_array_push(&node->items, bounds, value);
581 }

583 static bool qt_node_remove(qt_node_t* node, qt_value_t value)
584 {
585  QT_ASSERT(node);
586 
587  // Searches the items in this node and, if found, removes the item with the
588  // specified value
589  for (int i = 0; i < node->items.size; i++)
590  {
591  qt_item_t* item = &node->items.items[i];
592 
593  // If value is found, then remove the node from the items array
594  if (item->value == value)
595  {
596  qt_array_remove(&node->items, i);
597  return true;
598  }
599  }
600 
601  // If the item wasn't found, recursively search the subtrees of this node
602  for (int i = 0; i < 4; i++)
603  {
604  if (node->nodes[i])
605  {
606  if (qt_node_remove(node->nodes[i], value))
607  return true;
608  }
609  }
610 
611  // Value wasn't found
612  return false;
613 }

631 static void qt_node_query(const qt_node_t* node, const qt_rect_t* area, qt_array_t* array)
632 {
633  QT_ASSERT(node);
634  QT_ASSERT(area);
635  QT_ASSERT(array);
636 
637  // Searches for items in this node that intersect the area and adds them to
638  // the array
639  for (int i = 0; i < node->items.size; i++)
640  {
641  qt_item_t* item = &node->items.items[i];
642 
643  if (qt_rect_overlaps(area, &item->bounds))
644  qt_array_push(array, &item->bounds, item->value);
645  }
646 
647  // Loop over subtrees
648  for (int i = 0; i < 4; i++)
649  {
650  if (node->nodes[i])
651  {
652  // If the area contains the the entire subtree, all items in the
653  // subtree match and are recursively added to the array
654  if (qt_rect_contains(area, &node->bounds[i]))
655  {
656  qt_node_all_items(node->nodes[i], array);
657  }
658  else
659  {
660  // Otherwise, if the area intersects the bounds of the subtree,
661  // the subtree is recursively searched for items intersecting
662  // or contained within the area
663  if (qt_rect_overlaps(area, &node->bounds[i]))
664  {
665  qt_node_query(node->nodes[i], area, array);
666  }
667  }
668  }
669  }
