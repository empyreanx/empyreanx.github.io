703 pm_flt pm_lerp_angle(pm_flt angle1, pm_flt angle2, pm_flt alpha)
704 {
705  const pm_v2 v1 = pm_v2_make(pm_cos(angle1), pm_sin(angle1));
706  const pm_v2 v2 = pm_v2_make(pm_cos(angle2), pm_sin(angle2));
707 
708  // Calculuate cosine of angle between the two vectors
709  pm_flt dot = pm_clamp(pm_v2_dot(&v1, &v2), -1.0f, 1.0f);
710 
711  // LERP if the cosine is too close to its limits
712  if (pm_equal(dot, 1.0f) || pm_equal(dot, -1.0f))
713  {
714  pm_v2 tmp = pm_v2_lerp(&v1, &v2, alpha);
715  return pm_normalize_angle(pm_atan2(tmp.y, tmp.x));
716  }
717 
718  // Calculate angle
719  pm_flt angle = pm_acos(dot) * alpha;
720 
721  // Gram-Schmidt(construct a new vector 'v0' that is orthogonal to 'v1')
722  pm_v2 v0 = pm_v2_sub(v2, pm_v2_scale(v1, dot));
723  pm_v2_normalize(&v0);
724 
725  // Calcuate vector in new coordinate system
726  pm_v2 tmp1 = pm_v2_scale(v1, pm_cos(angle));
727  pm_v2 tmp2 = pm_v2_scale(v0, pm_sin(angle));
728 
729  pm_v2 tmp = pm_v2_add(tmp1, tmp2);
730 
731  // Calculate new angle
732  return pm_normalize_angle(pm_atan2(tmp.y, tmp.x));
733 }

913  pm_flt min_y = vertices[0].y;
914  pm_flt max_x = vertices[0].x;
915  pm_flt max_y = vertices[0].y;
916 
917  for (int i = 0; i < count; i++)
918  {
919  if (vertices[i].x < min_x)
920  min_x = vertices[i].x;
921 
922  if (vertices[i].y < min_x)
923  min_y = vertices[i].y;
924 
925  if (vertices[i].x > max_x)
926  max_x = vertices[i].x;
927 
928  if (vertices[i].y > max_y)

930  }

934  out.y = min_y;

936  out.h = pm_abs(max_y - min_y);

941 /*
942  * Mersenne Twister random number generator. Based on
943  * https://github.com/ESultanik/mtwister by Evan Sultanik (Public Domain).
944  */
945 
946 #define PM_STATE_VECTOR_LEN 624
947 #define PM_STATE_VECTOR_M 397
948 #define PM_UPPER_MASK 0x80000000
949 #define PM_LOWER_MASK 0x7fffffff
950 #define PM_TEMPERING_MASK_B 0x9d2c5680
951 #define PM_TEMPERING_MASK_C 0xefc60000
952 
953 void pm_rng_seed(pm_rng_t* rng, uint32_t seed)
954 {
955  rng->mt[0] = seed & 0xffffffff;
956 
957  for (rng->index = 1; rng->index < PM_STATE_VECTOR_LEN; rng->index++)
958  {
959  rng->mt[rng->index] = (69069 * rng->mt[rng->index - 1]) & 0xffffffff;
960  }
961 }
962 
963 uint32_t pm_random(pm_rng_t* rng)
964 {
965  uint32_t y;
966  static const uint32_t mag[2] = {0x0, 0x9908b0df};
967 
968  if (rng->index >= PM_STATE_VECTOR_LEN || rng->index < 0)
969  {
970  int kk;
971 
972  if (rng->index >= PM_STATE_VECTOR_LEN + 1 || rng->index < 0)
973  {
974  pm_rng_seed(rng, 4357);
975  }
976 
977  for (kk = 0; kk < PM_STATE_VECTOR_LEN - PM_STATE_VECTOR_M; kk++)
978  {
979  y = (rng->mt[kk] & PM_UPPER_MASK) | (rng->mt[kk + 1] & PM_LOWER_MASK);
980  rng->mt[kk] = rng->mt[kk + PM_STATE_VECTOR_M] ^ (y >> 1) ^ mag[y & 0x1];
981  }
982 
983  for(; kk < PM_STATE_VECTOR_LEN - 1; kk++)
984  {
985  y = (rng->mt[kk] & PM_UPPER_MASK) | (rng->mt[kk + 1] & PM_LOWER_MASK);
986  rng->mt[kk] = rng->mt[kk + (PM_STATE_VECTOR_M - PM_STATE_VECTOR_LEN)] ^ (y >> 1) ^ mag[y & 0x1];
987  }
988 
989  y = (rng->mt[PM_STATE_VECTOR_LEN - 1] & PM_UPPER_MASK) | (rng->mt[0] & PM_LOWER_MASK);
990 
991  rng->mt[PM_STATE_VECTOR_LEN - 1] = rng->mt[PM_STATE_VECTOR_M - 1] ^ (y >> 1) ^ mag[y & 0x1];
992  rng->index = 0;
993  }
994 
995  y = rng->mt[rng->index++];
996  y ^= (y >> 11);
997  y ^= (y << 7) & PM_TEMPERING_MASK_B;
998  y ^= (y << 15) & PM_TEMPERING_MASK_C;
999  y ^= (y >> 18);
1000 
1001  return y;
1002 }
1003 
1004 pm_flt pm_random_float(pm_rng_t* rng)
1005 {
1006  return (pm_flt)pm_random(rng) / (pm_flt)UINT32_MAX;
1007 }
