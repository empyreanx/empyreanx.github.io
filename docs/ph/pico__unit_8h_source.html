236bool
237pu_assert (bool passed,
238 const char* const expr,
239 const char* const file,
240 int line)
241{
242 pu_num_asserts++;
243
244 if (passed)
245 {
246 return true;
247 }
248
249 if (pu_colors)
250 {
251 printf("(%c%sFAILED%c%s: %s (%d): %s)\n",
252 TERM_COLOR_CODE, TERM_COLOR_RED,
253 TERM_COLOR_CODE, TERM_COLOR_RESET,
254 file, line, expr);
255 }
256 else
257 {
258 printf("(FAILED: %s (%d): %s)\n", file, line, expr);
259 }
260
261 return false;
262}

264void
265pu_run_test (const char* const name, pu_test_fn test_fp)
266{
267 if (NULL != pu_setup_fp)
268 {
269 pu_setup_fp();
270 }
271
272 printf("Running: %s ", name);
273
274 #ifndef PU_NO_CLOCK
275
276 clock_t start_time = 0;
277 clock_t end_time = 0;
278
279 if (pu_time)
280 {
281 start_time = clock();
282 }
283
284 #endif // PU_NO_CLOCK
285
286 if (!test_fp())
287 {
288 pu_num_failed++;
289
290 if (NULL != pu_teardown_fp)
291 {
292 pu_teardown_fp();
293 }
294
295 return;
296 }
297
298 #ifndef PU_NO_CLOCK
299
300 if (pu_time)
301 {
302 end_time = clock();
303 }
304
305 #endif // PU_NO_CLOCK
306
307 if (pu_colors)
308 {
309 printf("(%c%sOK%c%s)", TERM_COLOR_CODE, TERM_COLOR_GREEN,
310 TERM_COLOR_CODE, TERM_COLOR_RESET);
311 }
312 else
313 {
314 printf("(OK)");
315 }
316
317 #ifndef PU_NO_CLOCK
318
319 if (pu_time)
320 {
321 printf(" (%f secs)", (double)(end_time - start_time) / CLOCKS_PER_SEC);
322 }
323
324 #endif // PU_NO_CLOCK
325
326 printf("\n");
327
328 pu_num_passed++;
329
330 if (NULL != pu_teardown_fp)
331 {
332 pu_teardown_fp();
333 }
334}

336void
337pu_run_suite (const char* const name, pu_suite_fn suite_fp)
338{
339 printf("===============================================================\n");
340
341 if (pu_colors)
342 {
343 printf("%c%sRunning: %s%c%s\n", TERM_COLOR_CODE, TERM_COLOR_BOLD,
344 name,
345 TERM_COLOR_CODE, TERM_COLOR_RESET);
346 }
347 else
348 {
349 printf("Running: %s\n", name);
350 }
351
352 printf("---------------------------------------------------------------\n");
353 suite_fp();
354 pu_num_suites++;
355}

357void
358pu_print_stats (void)
359{
360 printf("===============================================================\n");
361
362 if (pu_colors)
363 {
364 printf("Summary: Passed: %c%s%u%c%s "
365 "Failed: %c%s%u%c%s "
366 "Total: %u Suites: %u "
367 "Asserts: %u\n",
368 TERM_COLOR_CODE, TERM_COLOR_GREEN, pu_num_passed,
369 TERM_COLOR_CODE, TERM_COLOR_RESET,
370 TERM_COLOR_CODE, TERM_COLOR_RED, pu_num_failed,
371 TERM_COLOR_CODE, TERM_COLOR_RESET,
372 pu_num_passed + pu_num_failed,
373 pu_num_suites, pu_num_asserts);
374 }
375 else
376 {
377 printf("Summary: Passed: %u "
378 "Failed: %u "
379 "Total: %u Suites: %u "
380 "Asserts: %u\n",
381 pu_num_passed,
382 pu_num_failed,
383 pu_num_passed + pu_num_failed,
384 pu_num_suites, pu_num_asserts);
385 }
