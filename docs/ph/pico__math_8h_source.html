811 pm_float pm_lerp_angle(pm_float angle1, pm_float angle2, pm_float alpha)
812 {
813  const pm_v2 v1 = pm_v2_make(pm_cos(angle1), pm_sin(angle1));
814  const pm_v2 v2 = pm_v2_make(pm_cos(angle2), pm_sin(angle2));
815 
816  // Calculuate cosine of angle between the two vectors
817  pm_float dot = pm_clamp(pm_v2_dot(v1, v2), -1.0f, 1.0f);
818 
819  // LERP if the cosine is too close to its limits
820  if (pm_equal(dot, 1.0f) || pm_equal(dot, -1.0f))
821  {
822  pm_v2 tmp = pm_v2_lerp(v1, v2, alpha);
823  return pm_normalize_angle(pm_atan2(tmp.y, tmp.x));
824  }
825 
826  // Calculate angle
827  pm_float angle = pm_acos(dot) * alpha;
828 
829  // Gram-Schmidt(construct a new vector 'v0' that is orthogonal to 'v1')
830  pm_v2 v0 = pm_v2_sub(v2, pm_v2_scale(v1, dot));
831  v0 = pm_v2_normalize(v0);
832 
833  // Calcuate vector in new coordinate system
834  pm_v2 tmp1 = pm_v2_scale(v1, pm_cos(angle));
835  pm_v2 tmp2 = pm_v2_scale(v0, pm_sin(angle));
836 
837  pm_v2 tmp = pm_v2_add(tmp1, tmp2);
838 
839  // Calculate new angle
840  return pm_normalize_angle(pm_atan2(tmp.y, tmp.x));
841 }

905 {

907 
908  if (0.0f == det) // Intentionally not using epsilon because determinants
909  { // can be really small and still be valid

921  out.ty = (t->t10 * t->tx - t->t00 * t->ty) * inv_det;

1036 /*
1037  * Mersenne Twister random number generator. Based on
1038  * https://github.com/ESultanik/mtwister by Evan Sultanik (Public Domain).
1039  */
1040 
1041 #define PM_STATE_VECTOR_LEN 624
1042 #define PM_STATE_VECTOR_M 397
1043 #define PM_UPPER_MASK 0x80000000
1044 #define PM_LOWER_MASK 0x7fffffff
1045 #define PM_TEMPERING_MASK_B 0x9d2c5680
1046 #define PM_TEMPERING_MASK_C 0xefc60000
1047 
1048 void pm_rng_seed(pm_rng_t* rng, uint32_t seed)
1049 {
1050  rng->mt[0] = seed & 0xffffffff;
1051 
1052  for (rng->index = 1; rng->index < PM_STATE_VECTOR_LEN; rng->index++)
1053  {
1054  rng->mt[rng->index] = (69069 * rng->mt[rng->index - 1]) & 0xffffffff;
1055  }
1056 }
1057 
1058 uint32_t pm_random(pm_rng_t* rng)
1059 {
1060  uint32_t y;
1061  static const uint32_t mag[2] = {0x0, 0x9908b0df};
1062 
1063  if (rng->index >= PM_STATE_VECTOR_LEN || rng->index < 0)
1064  {
1065  int kk;
1066 
1067  if (rng->index >= PM_STATE_VECTOR_LEN + 1 || rng->index < 0)
1068  {
1069  pm_rng_seed(rng, 4357);
1070  }
1071 
1072  for (kk = 0; kk < PM_STATE_VECTOR_LEN - PM_STATE_VECTOR_M; kk++)
1073  {
1074  y = (rng->mt[kk] & PM_UPPER_MASK) | (rng->mt[kk + 1] & PM_LOWER_MASK);
1075  rng->mt[kk] = rng->mt[kk + PM_STATE_VECTOR_M] ^ (y >> 1) ^ mag[y & 0x1];
1076  }
1077 
1078  for(; kk < PM_STATE_VECTOR_LEN - 1; kk++)
1079  {
1080  y = (rng->mt[kk] & PM_UPPER_MASK) | (rng->mt[kk + 1] & PM_LOWER_MASK);
1081  rng->mt[kk] = rng->mt[kk + (PM_STATE_VECTOR_M - PM_STATE_VECTOR_LEN)] ^ (y >> 1) ^ mag[y & 0x1];
1082  }
1083 
1084  y = (rng->mt[PM_STATE_VECTOR_LEN - 1] & PM_UPPER_MASK) | (rng->mt[0] & PM_LOWER_MASK);
1085 
1086  rng->mt[PM_STATE_VECTOR_LEN - 1] = rng->mt[PM_STATE_VECTOR_M - 1] ^ (y >> 1) ^ mag[y & 0x1];
1087  rng->index = 0;
1088  }
1089 
1090  y = rng->mt[rng->index++];
1091  y ^= (y >> 11);
1092  y ^= (y << 7) & PM_TEMPERING_MASK_B;
1093  y ^= (y << 15) & PM_TEMPERING_MASK_C;
1094  y ^= (y >> 18);
1095 
1096  return y;
1097 }
1098 
1099 pm_float pm_random_float(pm_rng_t* rng)
1100 {
1101  return (pm_float)pm_random(rng) / (pm_float)UINT32_MAX;
1102 }
