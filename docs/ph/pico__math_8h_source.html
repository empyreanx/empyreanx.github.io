822 pfloat pf_lerp_angle(pfloat angle1, pfloat angle2, pfloat alpha)
823 {
824  const pv2 v1 = pv2_make(pf_cos(angle1), pf_sin(angle1));
825  const pv2 v2 = pv2_make(pf_cos(angle2), pf_sin(angle2));
826 
827  // Calculuate cosine of angle between the two vectors
828  pfloat dot = pf_clamp(pv2_dot(v1, v2), -1.0f, 1.0f);
829 
830  // LERP if the cosine is too close to its limits
831  if (pf_equal(dot, 1.0f) || pf_equal(dot, -1.0f))
832  {
833  pv2 tmp = pv2_lerp(v1, v2, alpha);
834  return pf_normalize_angle(pf_atan2(tmp.y, tmp.x));
835  }
836 
837  // Calculate angle
838  pfloat angle = pf_acos(dot) * alpha;
839 
840  // Gram-Schmidt(construct a new vector 'v0' that is orthogonal to 'v1')
841  pv2 v0 = pv2_sub(v2, pv2_scale(v1, dot));
842  v0 = pv2_normalize(v0);
843 
844  // Calcuate vector in new coordinate system
845  pv2 tmp1 = pv2_scale(v1, pf_cos(angle));
846  pv2 tmp2 = pv2_scale(v0, pf_sin(angle));
847 
848  pv2 tmp = pv2_add(tmp1, tmp2);
849 
850  // Calculate new angle
851  return pf_normalize_angle(pf_atan2(tmp.y, tmp.x));
852 }

1037  verts[3] = pv2_make(pos.x + w, pos.y);

1040  verts[1] = pt2_map(t, verts[1]);

1048  * Implementation of the xoshiro128** algorithm
1049  * https://en.wikipedia.org/wiki/Xorshift
1050  */
1051 
1052 void prng_seed(prng_t* rng, uint64_t seed)
1053 {
1054  for (int i = 0; i < 2; i++)
1055  {
1056  uint64_t result = (seed += 0x9E3779B97f4A7C15);
1057  result = (result ^ (result >> 30)) * 0xBF58476D1CE4E5B9;
1058  result = (result ^ (result >> 27)) * 0x94D049BB133111EB;
1059  result = result ^ (result >> 31);
1060 
1061  int j = i * 2;
1062 
1063  rng->s[j] = (uint32_t)result;
1064  rng->s[j + 1] = (uint32_t)(result >> 32);
1065  }
1066 }
1067 
1068 static uint32_t rng_rol32(uint32_t x, int k)
1069 {
1070  return (x << k) | (x >> (32 - k));
1071 }
1072 
1073 uint32_t prng_random(prng_t* rng)
1074 {
1075  uint32_t *s = rng->s;
1076  uint32_t const result = rng_rol32(s[1] * 5, 7) * 9;
1077  uint32_t const t = s[1] << 17;
1078 
1079  s[2] ^= s[0];
1080  s[3] ^= s[1];
1081  s[1] ^= s[2];
1082  s[0] ^= s[3];
1083 
1084  s[2] ^= t;
1085  s[3] = rng_rol32(s[3], 45);
1086 
1087  return result;
1088 }
