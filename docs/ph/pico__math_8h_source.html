791 pm_flt pm_lerp_angle(pm_flt angle1, pm_flt angle2, pm_flt alpha)
792 {
793  const pm_v2 v1 = pm_v2_make(pm_cos(angle1), pm_sin(angle1));
794  const pm_v2 v2 = pm_v2_make(pm_cos(angle2), pm_sin(angle2));
795 
796  // Calculuate cosine of angle between the two vectors
797  pm_flt dot = pm_clamp(pm_v2_dot(&v1, &v2), -1.0f, 1.0f);
798 
799  // LERP if the cosine is too close to its limits
800  if (pm_equal(dot, 1.0f) || pm_equal(dot, -1.0f))
801  {
802  pm_v2 tmp = pm_v2_lerp(&v1, &v2, alpha);
803  return pm_normalize_angle(pm_atan2(tmp.y, tmp.x));
804  }
805 
806  // Calculate angle
807  pm_flt angle = pm_acos(dot) * alpha;
808 
809  // Gram-Schmidt(construct a new vector 'v0' that is orthogonal to 'v1')
810  pm_v2 v0 = pm_v2_sub(v2, pm_v2_scale(v1, dot));
811  pm_v2_normalize(&v0);
812 
813  // Calcuate vector in new coordinate system
814  pm_v2 tmp1 = pm_v2_scale(v1, pm_cos(angle));
815  pm_v2 tmp2 = pm_v2_scale(v0, pm_sin(angle));
816 
817  pm_v2 tmp = pm_v2_add(tmp1, tmp2);
818 
819  // Calculate new angle
820  return pm_normalize_angle(pm_atan2(tmp.y, tmp.x));
821 }

1016 /*
1017  * Mersenne Twister random number generator. Based on
1018  * https://github.com/ESultanik/mtwister by Evan Sultanik (Public Domain).
1019  */
1020 
1021 #define PM_STATE_VECTOR_LEN 624
1022 #define PM_STATE_VECTOR_M 397
1023 #define PM_UPPER_MASK 0x80000000
1024 #define PM_LOWER_MASK 0x7fffffff
1025 #define PM_TEMPERING_MASK_B 0x9d2c5680
1026 #define PM_TEMPERING_MASK_C 0xefc60000
1027 
1028 void pm_rng_seed(pm_rng_t* rng, uint32_t seed)
1029 {
1030  rng->mt[0] = seed & 0xffffffff;
1031 
1032  for (rng->index = 1; rng->index < PM_STATE_VECTOR_LEN; rng->index++)
1033  {
1034  rng->mt[rng->index] = (69069 * rng->mt[rng->index - 1]) & 0xffffffff;
1035  }
1036 }
1037 
1038 uint32_t pm_random(pm_rng_t* rng)
1039 {
1040  uint32_t y;
1041  static const uint32_t mag[2] = {0x0, 0x9908b0df};
1042 
1043  if (rng->index >= PM_STATE_VECTOR_LEN || rng->index < 0)
1044  {
1045  int kk;
1046 
1047  if (rng->index >= PM_STATE_VECTOR_LEN + 1 || rng->index < 0)
1048  {
1049  pm_rng_seed(rng, 4357);
1050  }
1051 
1052  for (kk = 0; kk < PM_STATE_VECTOR_LEN - PM_STATE_VECTOR_M; kk++)
1053  {
1054  y = (rng->mt[kk] & PM_UPPER_MASK) | (rng->mt[kk + 1] & PM_LOWER_MASK);
1055  rng->mt[kk] = rng->mt[kk + PM_STATE_VECTOR_M] ^ (y >> 1) ^ mag[y & 0x1];
1056  }
1057 
1058  for(; kk < PM_STATE_VECTOR_LEN - 1; kk++)
1059  {
1060  y = (rng->mt[kk] & PM_UPPER_MASK) | (rng->mt[kk + 1] & PM_LOWER_MASK);
1061  rng->mt[kk] = rng->mt[kk + (PM_STATE_VECTOR_M - PM_STATE_VECTOR_LEN)] ^ (y >> 1) ^ mag[y & 0x1];
1062  }
1063 
1064  y = (rng->mt[PM_STATE_VECTOR_LEN - 1] & PM_UPPER_MASK) | (rng->mt[0] & PM_LOWER_MASK);
1065 
1066  rng->mt[PM_STATE_VECTOR_LEN - 1] = rng->mt[PM_STATE_VECTOR_M - 1] ^ (y >> 1) ^ mag[y & 0x1];
1067  rng->index = 0;
1068  }
1069 
1070  y = rng->mt[rng->index++];
1071  y ^= (y >> 11);
1072  y ^= (y << 7) & PM_TEMPERING_MASK_B;
1073  y ^= (y << 15) & PM_TEMPERING_MASK_C;
1074  y ^= (y >> 18);
1075 
1076  return y;
1077 }
1078 
1079 pm_flt pm_random_float(pm_rng_t* rng)
1080 {
1081  return (pm_flt)pm_random(rng) / (pm_flt)UINT32_MAX;
1082 }
