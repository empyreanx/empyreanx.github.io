802 pm_float pm_lerp_angle(pm_float angle1, pm_float angle2, pm_float alpha)
803 {
804  const pm_v2 v1 = pm_v2_make(pm_cos(angle1), pm_sin(angle1));
805  const pm_v2 v2 = pm_v2_make(pm_cos(angle2), pm_sin(angle2));
806 
807  // Calculuate cosine of angle between the two vectors
808  pm_float dot = pm_clamp(pm_v2_dot(v1, v2), -1.0f, 1.0f);
809 
810  // LERP if the cosine is too close to its limits
811  if (pm_equal(dot, 1.0f) || pm_equal(dot, -1.0f))
812  {
813  pm_v2 tmp = pm_v2_lerp(v1, v2, alpha);
814  return pm_normalize_angle(pm_atan2(tmp.y, tmp.x));
815  }
816 
817  // Calculate angle
818  pm_float angle = pm_acos(dot) * alpha;
819 
820  // Gram-Schmidt(construct a new vector 'v0' that is orthogonal to 'v1')
821  pm_v2 v0 = pm_v2_sub(v2, pm_v2_scale(v1, dot));
822  v0 = pm_v2_normalize(v0);
823 
824  // Calcuate vector in new coordinate system
825  pm_v2 tmp1 = pm_v2_scale(v1, pm_cos(angle));
826  pm_v2 tmp2 = pm_v2_scale(v0, pm_sin(angle));
827 
828  pm_v2 tmp = pm_v2_add(tmp1, tmp2);
829 
830  // Calculate new angle
831  return pm_normalize_angle(pm_atan2(tmp.y, tmp.x));
832 }

1028  * Implementation of the xoshiro128** algorithm
1029  * https://en.wikipedia.org/wiki/Xorshift
1030  */
1031 
1032 void pm_rng_seed(pm_rng_t* rng, uint64_t seed)
1033 {
1034  for (int i = 0; i < 2; i++)
1035  {
1036  uint64_t result = (seed += 0x9E3779B97f4A7C15);
1037  result = (result ^ (result >> 30)) * 0xBF58476D1CE4E5B9;
1038  result = (result ^ (result >> 27)) * 0x94D049BB133111EB;
1039  result = result ^ (result >> 31);
1040 
1041  int j = i * 2;
1042 
1043  rng->s[j] = (uint32_t)result;
1044  rng->s[j + 1] = (uint32_t)(result >> 32);
1045  }
1046 }
1047 
1048 static uint32_t rng_rol32(uint32_t x, int k)
1049 {
1050  return (x << k) | (x >> (32 - k));
1051 }
1052 
1053 uint32_t pm_random(pm_rng_t* rng)
1054 {
1055  uint32_t *s = rng->s;
1056  uint32_t const result = rng_rol32(s[1] * 5, 7) * 9;
1057  uint32_t const t = s[1] << 17;
1058 
1059  s[2] ^= s[0];
1060  s[3] ^= s[1];
1061  s[1] ^= s[2];
1062  s[0] ^= s[3];
1063 
1064  s[2] ^= t;
1065  s[3] = rng_rol32(s[3], 45);
1066 
1067  return result;
1068 }

1071 {
1072  return (pm_float)pm_random(rng) / (pm_float)UINT32_MAX;
