794 pm_float pm_lerp_angle(pm_float angle1, pm_float angle2, pm_float alpha)
795 {
796  const pm_v2 v1 = pm_v2_make(pm_cos(angle1), pm_sin(angle1));
797  const pm_v2 v2 = pm_v2_make(pm_cos(angle2), pm_sin(angle2));
798 
799  // Calculuate cosine of angle between the two vectors
800  pm_float dot = pm_clamp(pm_v2_dot(v1, v2), -1.0f, 1.0f);
801 
802  // LERP if the cosine is too close to its limits
803  if (pm_equal(dot, 1.0f) || pm_equal(dot, -1.0f))
804  {
805  pm_v2 tmp = pm_v2_lerp(v1, v2, alpha);
806  return pm_normalize_angle(pm_atan2(tmp.y, tmp.x));
807  }
808 
809  // Calculate angle
810  pm_float angle = pm_acos(dot) * alpha;
811 
812  // Gram-Schmidt(construct a new vector 'v0' that is orthogonal to 'v1')
813  pm_v2 v0 = pm_v2_sub(v2, pm_v2_scale(v1, dot));
814  v0 = pm_v2_normalize(v0);
815 
816  // Calcuate vector in new coordinate system
817  pm_v2 tmp1 = pm_v2_scale(v1, pm_cos(angle));
818  pm_v2 tmp2 = pm_v2_scale(v0, pm_sin(angle));
819 
820  pm_v2 tmp = pm_v2_add(tmp1, tmp2);
821 
822  // Calculate new angle
823  return pm_normalize_angle(pm_atan2(tmp.y, tmp.x));
824 }

1019 /*
1020  * Mersenne Twister random number generator. Based on
1021  * https://github.com/ESultanik/mtwister by Evan Sultanik (Public Domain).
1022  */
1023 
1024 #define PM_STATE_VECTOR_LEN 624
1025 #define PM_STATE_VECTOR_M 397
1026 #define PM_UPPER_MASK 0x80000000
1027 #define PM_LOWER_MASK 0x7fffffff
1028 #define PM_TEMPERING_MASK_B 0x9d2c5680
1029 #define PM_TEMPERING_MASK_C 0xefc60000
1030 
1031 void pm_rng_seed(pm_rng_t* rng, uint32_t seed)
1032 {
1033  rng->mt[0] = seed & 0xffffffff;
1034 
1035  for (rng->index = 1; rng->index < PM_STATE_VECTOR_LEN; rng->index++)
1036  {
1037  rng->mt[rng->index] = (69069 * rng->mt[rng->index - 1]) & 0xffffffff;
1038  }
1039 }
1040 
1041 uint32_t pm_random(pm_rng_t* rng)
1042 {
1043  uint32_t y;
1044  static const uint32_t mag[2] = {0x0, 0x9908b0df};
1045 
1046  if (rng->index >= PM_STATE_VECTOR_LEN || rng->index < 0)
1047  {
1048  int kk;
1049 
1050  if (rng->index >= PM_STATE_VECTOR_LEN + 1 || rng->index < 0)
1051  {
1052  pm_rng_seed(rng, 4357);
1053  }
1054 
1055  for (kk = 0; kk < PM_STATE_VECTOR_LEN - PM_STATE_VECTOR_M; kk++)
1056  {
1057  y = (rng->mt[kk] & PM_UPPER_MASK) | (rng->mt[kk + 1] & PM_LOWER_MASK);
1058  rng->mt[kk] = rng->mt[kk + PM_STATE_VECTOR_M] ^ (y >> 1) ^ mag[y & 0x1];
1059  }
1060 
1061  for(; kk < PM_STATE_VECTOR_LEN - 1; kk++)
1062  {
1063  y = (rng->mt[kk] & PM_UPPER_MASK) | (rng->mt[kk + 1] & PM_LOWER_MASK);
1064  rng->mt[kk] = rng->mt[kk + (PM_STATE_VECTOR_M - PM_STATE_VECTOR_LEN)] ^ (y >> 1) ^ mag[y & 0x1];
1065  }
1066 
1067  y = (rng->mt[PM_STATE_VECTOR_LEN - 1] & PM_UPPER_MASK) | (rng->mt[0] & PM_LOWER_MASK);
1068 
1069  rng->mt[PM_STATE_VECTOR_LEN - 1] = rng->mt[PM_STATE_VECTOR_M - 1] ^ (y >> 1) ^ mag[y & 0x1];
1070  rng->index = 0;
1071  }
1072 
1073  y = rng->mt[rng->index++];
1074  y ^= (y >> 11);
1075  y ^= (y << 7) & PM_TEMPERING_MASK_B;
1076  y ^= (y << 15) & PM_TEMPERING_MASK_C;
1077  y ^= (y >> 18);
1078 
1079  return y;
1080 }

1082 pm_float pm_random_float(pm_rng_t* rng)
1083 {
1084  return (pm_float)pm_random(rng) / (pm_float)UINT32_MAX;
1085 }
