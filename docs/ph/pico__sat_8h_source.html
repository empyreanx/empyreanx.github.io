253 sat_poly_t sat_make_poly(pm_v2 vertices[], int vertex_count)
254 {
255  SAT_ASSERT(vertex_count <= PICO_SAT_MAX_POLY_VERTS);
256  SAT_ASSERT(vertices);
257 
258  sat_poly_t poly;
259 
260  // Copy vertices
261  poly.vertex_count = vertex_count;
262 
263  for (int i = 0; i < vertex_count; i++)
264  {
265  poly.vertices[i] = vertices[i];
266  }
267 
268  // Cache edges and edge normals
269  for (int i = 0; i < vertex_count; i++)
270  {
271  int next = (i + 1) == vertex_count ? 0 : i + 1;
272 
273  pm_v2 v1 = vertices[i];
274  pm_v2 v2 = vertices[next];
275  poly.edges[i] = pm_v2_sub(v2, v1);
276  poly.normals[i] = pm_v2_perp(poly.edges[i]);
277  poly.normals[i] = pm_v2_normalize(poly.normals[i]);
278  }
279 
280  return poly;
281 }

311  sat_init_manifold(manifold);
312 
313  // Test axises on poly1
314  for (int i = 0; i < poly1->vertex_count; i++)
315  {
316  // Get signed overlap of poly1 on poly2 along specified normal direction
317  pm_float overlap = sat_axis_overlap(poly1, poly2, poly1->normals[i]);

346 bool sat_test_poly_circle(const sat_poly_t* poly,
347  const sat_circle_t* circle,
348  sat_manifold_t* manifold)
349 {
350  SAT_ASSERT(poly);
351  SAT_ASSERT(circle);
352 
353  if (manifold)
354  sat_init_manifold(manifold);
355 
356  pm_float radius2 = circle->radius * circle->radius;
357 
358  // The main idea behind this function is to classify the position of the
359  // circle relative to the Voronoi region(s) it is part of. This uses very
360  // inexpensive operations. Essentially it efficiently narrows down the
361  // position of the circle so that the correct separating axis test can be
362  // performed.
363 
364  int count = poly->vertex_count;
365 
366  for (int i = 0; i < count; i++)
367  {
368  int next = (i + 1) == count ? 0 : i + 1;
369  int prev = (i - 1) <= 0 ? count - 1 : i - 1;
370 
371  // Edge to test
372  pm_v2 edge = poly->edges[i];
373 
374  // Postion of circle relative to vertex
375  pm_v2 point = pm_v2_sub(circle->pos, poly->vertices[i]);
376 
377  // Find the Voronoi region of the point (circle center relative to
378  // vertex) with respect to the edge
379  sat_voronoi_region_t region = sat_voronoi_region(point, edge);
380 
381  // Test if point is in the left Voronoi region
382  if (region == SAT_VORONOI_LEFT)
383  {
384  // If it is, check if it is in the right Voronoi region of the
385  // previous edge. If this is the case, the circle is "sandwiched"
386  // in the intersection of the Voronoi regions defined by the
387  // endpoints of the edges.
388 
389  pm_v2 point2 = pm_v2_sub(circle->pos, poly->vertices[prev]);
390  edge = poly->edges[prev];
391 
392  region = sat_voronoi_region(point2, edge);
393 
394  if (region == SAT_VORONOI_RIGHT)
395  {
396  // The circle center is in the left/right Voronoi region, so
397  // check to see if it contains the vertex
398  pm_float diff2 = pm_v2_len2(point);
399 
400  // Equivalent to diff > radius
401  if (diff2 > radius2)
402  return false;
403 
404  // Vertex is contained within circle, so the circle overlaps the
405  // polygon
406 
407  if (manifold)
408  {
409  // Calculate distance because we need it now
410  pm_float diff = pm_sqrt(diff2);
411 
412  // Calculate overlap
413  pm_float overlap = circle->radius - diff;
414 
415  // Normal direction is just the circle relative to the vertex
416  pm_v2 normal = pm_v2_normalize(point);
417 
418  // Update manifold
419  sat_update_manifold(manifold, normal, overlap);
420  }
421  }
422  }
423  // This case is symmetric to the above
424  else if (region == SAT_VORONOI_RIGHT)
425  {
426  pm_v2 point2 = pm_v2_sub(circle->pos, poly->vertices[next]);
427  edge = poly->edges[next];
428 
429  region = sat_voronoi_region(point2, edge);
430 
431  if (region == SAT_VORONOI_LEFT)
432  {
433  pm_float diff2 = pm_v2_len2(point);
434 
435  if (diff2 > radius2)
436  return false;
437 
438  if (manifold)
439  {
440  pm_float diff = pm_sqrt(diff2);
441  pm_float overlap = circle->radius - diff;
442  pm_v2 normal = pm_v2_normalize(point);
443  sat_update_manifold(manifold, normal, overlap);
444  }
445  }
446  }
447  else // SAT_VORONOI_MIDDLE
448  {
449  // In this case, the location of the circle is between the endpoints
450  // of an edge.
451  pm_v2 normal = poly->normals[i];
452 
453  // Location of center of circle along the edge normal
454  pm_float diff = pm_v2_dot(normal, point);
455  pm_float abs_diff = pm_abs(diff);
456 
457  // Test if circle does not intersect edge
458  if (diff > 0.0f && abs_diff > circle->radius)
459  return false;
460 
461  if (manifold)
462  {
463  // Calculate overlap
464  pm_float overlap = circle->radius - diff;
465 
466  // Update manifold
467  sat_update_manifold(manifold, normal, overlap);
468  }
469  }
470  }
471 
472  return true;
473 }

495 bool sat_test_circle_circle(const sat_circle_t* circle1,
496  const sat_circle_t* circle2,
497  sat_manifold_t* manifold)
498 {
499  SAT_ASSERT(circle1);
500  SAT_ASSERT(circle2);
501 
502  if (manifold)
503  sat_init_manifold(manifold);
504 
505  // Position of circle2 relative to circle1
506  pm_v2 diff = pm_v2_sub(circle2->pos, circle1->pos);
507 
508  // Squared distance between circle centers
509  pm_float dist2 = pm_v2_len2(diff);
510 
511  // Sum of radii
512  pm_float total_radius = circle1->radius + circle2->radius;
513 
514  // Square sum of radii for optimization (avoid calculating length until we
515  // have to)
516  pm_float total_radius2 = total_radius * total_radius;
517 
518  // Equivalent to dist >= total_radius
519  if (dist2 >= total_radius2)
520  return false;
521 
522  if (manifold)
523  {
524  // Calculate distance because we need it now
525  pm_float dist = pm_sqrt(dist2);
526 
527  // Calculate overlap
528  pm_float overlap = total_radius - dist;
529 
530  // Normal direction is just circle2 relative to circle1
531  pm_v2 normal = pm_v2_normalize(diff);
532 
533  // Update manifold
534  sat_update_manifold(manifold, normal, overlap);
535  }
536 
537  return true;

553 static void sat_update_manifold(sat_manifold_t* manifold, pm_v2 normal, pm_float overlap)
554 {
555  SAT_ASSERT(manifold);
556 
557  pm_float abs_overlap = pm_abs(overlap);
558 
559  // Only update if the new overlap is smaller than the old one
560  if (abs_overlap < manifold->overlap)
561  {
562  // Update overlap (always positive)
563  manifold->overlap = abs_overlap;
564 
565  // If the overlap is less that zero the normal must be reversed
566  if (overlap < 0.0f)
567  manifold->normal = pm_v2_neg(normal);
568  else if (overlap > 0.0f)
569  manifold->normal = normal;
570 
571  manifold->vector = pm_v2_scale(manifold->normal, manifold->overlap);
572  }
573 }

575 static void sat_axis_range(const sat_poly_t* poly, pm_v2 normal, pm_float range[2])
576 {
577  SAT_ASSERT(poly);
578  SAT_ASSERT(range);
579 
580  pm_float dot = pm_v2_dot(poly->vertices[0], normal);
581  pm_float min = dot;
582  pm_float max = dot;
583 
584  // Find the minimum and maximum distance of the polygon along the normal
585  for (int i = 1; i < poly->vertex_count; i++)
586  {
587  dot = pm_v2_dot(poly->vertices[i], normal);
588 
589  if (dot < min)
590  min = dot;
591 
592  if (dot > max)
593  max = dot;
594  }
595 
596  // The range defines the interval induced by the polygon projected onto the
597  // normal vector
598  range[0] = min;
599  range[1] = max;
600 }

602 static pm_float sat_axis_overlap(const sat_poly_t* poly1,
603  const sat_poly_t* poly2,
604  pm_v2 axis)
605 
606 {
607  SAT_ASSERT(poly1);
608  SAT_ASSERT(poly2);
609 
610  pm_float range1[2];
611  pm_float range2[2];
612 
613  // Get the ranges of polygons projected onto the axis vector
614  sat_axis_range(poly1, axis, range1);
615  sat_axis_range(poly2, axis, range2);
616 
617  // Ranges do not overlaps
618  if (range1[1] < range2[0] || range2[1] < range1[0])
619  return 0.0f;
620 
621  // Calculate overlap candidates
622  pm_float overlap1 = range1[1] - range2[0];
623  pm_float overlap2 = range2[1] - range1[0];
624 
625  // Return the smaller overlap
626  return (overlap2 > overlap1) ? overlap1 : -overlap2;
627 }

629 static sat_voronoi_region_t sat_voronoi_region(pm_v2 point, pm_v2 line)
630 {
631  pm_float len2 = pm_v2_len2(line);
632  pm_float dot = pm_v2_dot(point, line);
633 
634  if (dot < 0.0f) // Point is to the left of the line
635  return SAT_VORONOI_LEFT;
636  else if (dot > len2) // Point is to the right of the line
637  return SAT_VORONOI_RIGHT;
638  else
639  return SAT_VORONOI_MIDDLE; // Point is somewhere in the middle
640 }
